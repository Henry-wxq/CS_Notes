\documentclass{article}

\usepackage{enumerate}
\usepackage{longtable}

\begin{document}

% 使用itemize默认列表（不需要导入宏包）
\begin{itemize}
	\item this is item 1
	% 注意嵌套时，\item的位置
	\begin{itemize}
	\item this is a subitemize
	\end{itemize}
	\item this is item 2
	\item this is item 3
\end{itemize}

\begin{enumerate}
	\item this is item 1
	% 注意嵌套时，\item的位置
	\begin{enumerate}
	\item this is item a
	\end{enumerate}
	\item this is item 2
	\item this is item 3
\end{enumerate}
	
% 设置前面的引导符号，使用中括号;bfseries可以设置加粗
\begin{enumerate}[\bfseries A.]
	\item this is item 1
	% 注意嵌套时，\item的位置
	\begin{enumerate}[\bfseries a.]
	\item this is item a
	\end{enumerate}
	\item this is item 2
	\item this is item 3
\end{enumerate}

% 想从f开始计数；前面要加上计数器；i表示第一层；ii表示第二层
% 就算是字母，也要用数字表示第几个字母
\begin{enumerate}[\bfseries A.]
\setcounter{enumi} {4}
	\item this is item 1
	% 注意嵌套时，\item的位置
	\begin{enumerate}[\bfseries a.]
	\item this is item a
	\end{enumerate}
	\item this is item 2
	\item this is item 3
\end{enumerate}

% 换页
\eject

% 表格如下：l:left; c: center; r:right
% |:竖线；hline:横线
% \\表示表格中的行间区分
\begin{tabular}{|l|c|r|}
	\hline
	1&20&300\\
	\hline
	200&10&3\\
	\hline
\end{tabular}

% 从excel表格导入到latex中时，需要把tab替换成&即可
% 可使用pages中的edit-查找和替换










\end{document}
